07 · Research & Open Source

An open developer ecosystem for hybrid quantum.

Use familiar Python tooling, our Ocean-inspired SDK, and high-level quantum workflow APIs to ship optimization workloads to the ObsidianQ cloud in a single import.

routing.py · ObsidianQ SDK 
# Hybrid quantum routing — 4500-qubit dispatch
from obsidianq import HybridSolver, Problem
from obsidianq.industrial import RoutingModel

problem = Problem.from_yaml("fleet.yaml")
model   = RoutingModel(
    vehicles=320,
    stops   =12_400,
    horizon ="24h",
    objective="min_total_distance",
)

solver  = HybridSolver(qpu="obsidian-pegasus", classical="gpu-mesh")
result  = solver.solve(model, time_limit=15)

print(result.kpis)   # > {"distance": 18421.2, "co2": 4.18t, "sla": 1.0}
